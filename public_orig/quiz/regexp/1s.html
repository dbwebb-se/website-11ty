#\b(?]+(?:\([\w\d]+\)|([^[:punct:]\s]|/))#

public function makeClickable($text)
{
    return preg_replace_callback(
        '#\b(?]+(?:\([\w\d]+\)|([^[:punct:]\s]|/))#',
        function ($matches) {
            return "{$matches[0]}";
        },
        $text
    );
}
